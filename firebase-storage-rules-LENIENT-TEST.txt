// Use these temporarily to see if rules are causing the upload error

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow everything for authenticated users (VERY PERMISSIVE)
    match /{allPaths=**} {
      allow read, write: if request.auth != null;

  }
}
